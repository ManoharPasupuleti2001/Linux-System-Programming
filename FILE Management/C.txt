28)Toggle a given range of bits
For example, take the number 245. The equivalent binary format is 11110101and the range is 4 to 7. So, the output should be 00001010 which is 5 in decimal.
CODE:
1)#include<stdio.h>
int main()
{
int num,start=4,end=7;
printf("Enter any number:");
scanf("%d",&num);
int mask=((1<<(end-start+1))-1)<<start;
printf("The given Value is %d\n",num^mask);
}
2)
#include<stdio.h>
int main()
{
unsigned char num=245;
unsigned char mask = 0xf0;
printf("after toggle bits of 4 and 7 :%d",num ^ mask);
return 0;
}

29)How to check if a particular bit is set or not in a number?
#include<stdio.h>
int main()
{
int num=245;
int pos,val;
printf("Enter the POS :");
scanf("%d",&pos);
val = (num & (1<<pos));
printf("\nval :%d\n",val);
if(val)
printf("num :%d Pos :%d val:%d is a set bit\n",num,pos,val);
else
printf("num :%d Pos :%d val:%d is not a set bit\n",num,pos,val);
return 0;
}
